<?php
/**
 * Database Connection Manager
 */
class Database {
    private static $instance = null;
    private $pdo;
    
    private function __construct() {
        try {
            $this->pdo = new PDO(
                "mysql:host=" . DB_HOST . ";dbname=" . DB_NAME . ";charset=utf8mb4",
                DB_USER,
                DB_PASS,
                [
                    PDO::ATTR_ERRMODE => PDO::ERRMODE_EXCEPTION,
                    PDO::ATTR_DEFAULT_FETCH_MODE => PDO::FETCH_ASSOC,
                    PDO::ATTR_EMULATE_PREPARES => false,
                ]
            );
        } catch (PDOException $e) {
            if (DEBUG_MODE) {
                die("Database connection failed: " . $e->getMessage());
            }
            die("Database connection failed");
        }
    }
    
    public static function getInstance(): self {
        if (self::$instance === null) {
            self::$instance = new self();
        }
        return self::$instance;
    }
    
    public function getConnection(): PDO {
        return $this->pdo;
    }
    
    public function query(string $sql, array $params = []): PDOStatement {
        $stmt = $this->pdo->prepare($sql);
        $stmt->execute($params);
        return $stmt;
    }
    
    public function fetch(string $sql, array $params = []): ?array {
        return $this->query($sql, $params)->fetch() ?: null;
    }
    
    public function fetchAll(string $sql, array $params = []): array {
        return $this->query($sql, $params)->fetchAll();
    }
    
    public function insert(string $table, array $data): int {
        $columns = implode(', ', array_keys($data));
        $placeholders = implode(', ', array_fill(0, count($data), '?'));
        
        $sql = "INSERT INTO `$table` ($columns) VALUES ($placeholders)";
        $this->query($sql, array_values($data));
        
        return (int)$this->pdo->lastInsertId();
    }
    
    public function update(string $table, array $data, string $where, array $whereParams = []): int {
        $set = implode(' = ?, ', array_keys($data)) . ' = ?';
        $sql = "UPDATE `$table` SET $set WHERE $where";
        
        $stmt = $this->query($sql, array_merge(array_values($data), $whereParams));
        return $stmt->rowCount();
    }
    
    public function delete(string $table, string $where, array $params = []): int {
        $sql = "DELETE FROM `$table` WHERE $where";
        return $this->query($sql, $params)->rowCount();
    }
    
    public function getSetting(string $key, $default = null) {
        $result = $this->fetch(
            "SELECT setting_value FROM settings WHERE setting_key = ?",
            [$key]
        );
        return $result ? $result['setting_value'] : $default;
    }
    
    public function setSetting(string $key, $value): void {
        $this->query(
            "INSERT INTO settings (setting_key, setting_value) VALUES (?, ?) 
             ON DUPLICATE KEY UPDATE setting_value = VALUES(setting_value)",
            [$key, $value]
        );
    }
    
    public function getAllSettings(): array {
        $results = $this->fetchAll("SELECT setting_key, setting_value FROM settings");
        $settings = [];
        foreach ($results as $row) {
            $settings[$row['setting_key']] = $row['setting_value'];
        }
        return $settings;
    }
}

<?php
/**
 * Authentication Manager for Dashboard
 */
class Auth {
    private $db;
    
    public function __construct() {
        $this->db = Database::getInstance();
        if (session_status() === PHP_SESSION_NONE) {
            session_start();
        }
    }
    
    public function login(string $username, string $password): array {
        $admin = $this->db->fetch(
            "SELECT * FROM admins WHERE username = ?",
            [$username]
        );
        
        if (!$admin) {
            Logger::log('warning', 'login_failed', "Invalid username: $username");
            return ['success' => false, 'error' => 'Неверные учетные данные'];
        }
        
        if (!password_verify($password, $admin['password_hash'])) {
            Logger::log('warning', 'login_failed', "Invalid password for: $username");
            return ['success' => false, 'error' => 'Неверные учетные данные'];
        }
        
        // Обновляем время входа
        $this->db->update('admins', ['last_login' => date('Y-m-d H:i:s')], 'id = ?', [$admin['id']]);
        
        // Создаем сессию
        $_SESSION['admin_id'] = $admin['id'];
        $_SESSION['admin_username'] = $admin['username'];
        $_SESSION['admin_role'] = $admin['role'];
        $_SESSION['login_time'] = time();
        
        // Генерируем CSRF токен
        $_SESSION['csrf_token'] = bin2hex(random_bytes(32));
        
        Logger::log('info', 'login_success', "Admin logged in: $username");
        
        return ['success' => true];
    }
    
    public function logout(): void {
        if (isset($_SESSION['admin_username'])) {
            Logger::log('info', 'logout', "Admin logged out: " . $_SESSION['admin_username']);
        }
        
        $_SESSION = [];
        if (ini_get("session.use_cookies")) {
            $params = session_get_cookie_params();
            setcookie(session_name(), '', time() - 42000,
                $params["path"], $params["domain"],
                $params["secure"], $params["httponly"]
            );
        }
        session_destroy();
    }
    
    public function isLoggedIn(): bool {
        if (!isset($_SESSION['admin_id'])) {
            return false;
        }
        
        // Проверяем время сессии (8 часов)
        if (time() - ($_SESSION['login_time'] ?? 0) > 28800) {
            $this->logout();
            return false;
        }
        
        return true;
    }
    
    public function requireLogin(): void {
        if (!$this->isLoggedIn()) {
            header('Location: /dashboard/login.php');
            exit;
        }
    }
    
    public function requireRole(string $role): void {
        $this->requireLogin();
        
        if ($_SESSION['admin_role'] !== $role && $_SESSION['admin_role'] !== 'admin') {
            http_response_code(403);
            die('Access denied');
        }
    }
    
    public function validateCSRF(string $token): bool {
        return isset($_SESSION['csrf_token']) && hash_equals($_SESSION['csrf_token'], $token);
    }
    
    public function getCSRFToken(): string {
        if (!isset($_SESSION['csrf_token'])) {
            $_SESSION['csrf_token'] = bin2hex(random_bytes(32));
        }
        return $_SESSION['csrf_token'];
    }
    
    public function changePassword(int $adminId, string $currentPassword, string $newPassword): array {
        $admin = $this->db->fetch("SELECT password_hash FROM admins WHERE id = ?", [$adminId]);
        
        if (!$admin || !password_verify($currentPassword, $admin['password_hash'])) {
            return ['success' => false, 'error' => 'Неверный текущий пароль'];
        }
        
        if (strlen($newPassword) < 8) {
            return ['success' => false, 'error' => 'Пароль должен быть минимум 8 символов'];
        }
        
        $hash = password_hash($newPassword, PASSWORD_ARGON2ID);
        $this->db->update('admins', ['password_hash' => $hash], 'id = ?', [$adminId]);
        
        Logger::log('info', 'password_changed', "Password changed for admin ID: $adminId");
        
        return ['success' => true];
    }
}

<?php
/**
 * Simple JWT Implementation
 */
class JWT {
    private static function base64UrlEncode(string $data): string {
        return rtrim(strtr(base64_encode($data), '+/', '-_'), '=');
    }
    
    private static function base64UrlDecode(string $data): string {
        return base64_decode(strtr($data, '-_', '+/'));
    }
    
    public static function encode(array $payload, int $expiry = 3600): string {
        $header = ['alg' => 'HS256', 'typ' => 'JWT'];
        
        $payload['iat'] = time();
        $payload['exp'] = time() + $expiry;
        
        $headerEncoded = self::base64UrlEncode(json_encode($header));
        $payloadEncoded = self::base64UrlEncode(json_encode($payload));
        
        $signature = hash_hmac('sha256', "$headerEncoded.$payloadEncoded", JWT_SECRET, true);
        $signatureEncoded = self::base64UrlEncode($signature);
        
        return "$headerEncoded.$payloadEncoded.$signatureEncoded";
    }
    
    public static function decode(string $token): ?array {
        $parts = explode('.', $token);
        if (count($parts) !== 3) {
            return null;
        }
        
        [$headerEncoded, $payloadEncoded, $signatureEncoded] = $parts;
        
        // Проверяем подпись
        $signature = self::base64UrlDecode($signatureEncoded);
        $expectedSignature = hash_hmac('sha256', "$headerEncoded.$payloadEncoded", JWT_SECRET, true);
        
        if (!hash_equals($expectedSignature, $signature)) {
            return null;
        }
        
        $payload = json_decode(self::base64UrlDecode($payloadEncoded), true);
        
        if (!$payload) {
            return null;
        }
        
        // Проверяем срок действия
        if (isset($payload['exp']) && $payload['exp'] < time()) {
            return null;
        }
        
        return $payload;
    }
    
    public static function generateSessionToken(): string {
        return bin2hex(random_bytes(32));
    }
    
    public static function generateVerificationToken(): string {
        return bin2hex(random_bytes(16));
    }
}

<?php
/**
 * Key Generation and Validation Manager
 */
class KeyManager {
    private $db;
    private $settings;
    
    public function __construct() {
        $this->db = Database::getInstance();
        $this->settings = $this->db->getAllSettings();
    }
    
    public function generateKey(string $hwid, string $ip, int $sessionId = null, bool $isPremium = false): string {
        $prefix = $this->settings['key_prefix'] ?? DEFAULT_KEY_PREFIX;
        $duration = (int)($this->settings['key_duration'] ?? DEFAULT_KEY_DURATION);
        
        // Деактивируем старые ключи для этого HWID
        $this->db->update(
            'keys',
            ['is_active' => false],
            'hwid = ? AND is_active = 1',
            [$hwid]
        );
        
        // Генерируем уникальный ключ
        $keyParts = [
            $prefix,
            strtoupper(substr(md5($hwid . time()), 0, 4)),
            strtoupper(bin2hex(random_bytes(4))),
            strtoupper(substr(md5(uniqid(mt_rand(), true)), 0, 4)),
            strtoupper(bin2hex(random_bytes(4)))
        ];
        
        $keyValue = implode('-', $keyParts);
        
        // Проверяем уникальность
        while ($this->db->fetch("SELECT id FROM keys WHERE key_value = ?", [$keyValue])) {
            $keyParts[2] = strtoupper(bin2hex(random_bytes(4)));
            $keyValue = implode('-', $keyParts);
        }
        
        // Сохраняем ключ
        $expiresAt = date('Y-m-d H:i:s', time() + $duration);
        
        $this->db->insert('keys', [
            'key_value' => $keyValue,
            'hwid' => $hwid,
            'ip_address' => $ip,
            'session_id' => $sessionId,
            'is_premium' => $isPremium ? 1 : 0,
            'is_active' => 1,
            'expires_at' => $expiresAt
        ]);
        
        // Обновляем статистику
        $this->updateStats('keys_generated');
        
        Logger::log('info', 'key_generated', "Key generated for HWID: " . substr($hwid, 0, 16) . "...", [
            'hwid' => $hwid,
            'ip' => $ip,
            'key_prefix' => $prefix
        ]);
        
        return $keyValue;
    }
    
    public function validateKey(string $key, string $hwid): array {
        $keyData = $this->db->fetch(
            "SELECT * FROM `keys` WHERE key_value = ?",
            [$key]
        );
        
        if (!$keyData) {
            Logger::log('warning', 'key_invalid', "Invalid key attempt", ['key' => substr($key, 0, 20)]);
            return [
                'valid' => false,
                'error' => 'Invalid key',
                'code' => 'KEY_NOT_FOUND'
            ];
        }
        
        if (!$keyData['is_active']) {
            return [
                'valid' => false,
                'error' => 'Key is deactivated',
                'code' => 'KEY_DEACTIVATED'
            ];
        }
        
        if (strtotime($keyData['expires_at']) < time()) {
            // Деактивируем просроченный ключ
            $this->db->update('keys', ['is_active' => false], 'id = ?', [$keyData['id']]);
            return [
                'valid' => false,
                'error' => 'Key has expired',
                'code' => 'KEY_EXPIRED'
            ];
        }
        
        if ($keyData['hwid'] !== $hwid) {
            Logger::log('warning', 'key_hwid_mismatch', "HWID mismatch for key", [
                'key' => substr($key, 0, 20),
                'expected_hwid' => substr($keyData['hwid'], 0, 16),
                'provided_hwid' => substr($hwid, 0, 16)
            ]);
            return [
                'valid' => false,
                'error' => 'Key is bound to different device',
                'code' => 'HWID_MISMATCH'
            ];
        }
        
        // Обновляем статистику использования
        $this->db->update('keys', [
            'last_used_at' => date('Y-m-d H:i:s'),
            'use_count' => $keyData['use_count'] + 1
        ], 'id = ?', [$keyData['id']]);
        
        $this->updateStats('keys_validated');
        
        return [
            'valid' => true,
            'expires_at' => $keyData['expires_at'],
            'expires_in' => strtotime($keyData['expires_at']) - time(),
            'is_premium' => (bool)$keyData['is_premium']
        ];
    }
    
    public function getKeyInfo(string $key): ?array {
        return $this->db->fetch(
            "SELECT key_value, is_premium, is_active, created_at, expires_at, use_count 
             FROM `keys` WHERE key_value = ?",
            [$key]
        );
    }
    
    public function getKeyByHWID(string $hwid): ?array {
        return $this->db->fetch(
            "SELECT * FROM `keys` 
             WHERE hwid = ? AND is_active = 1 AND expires_at > NOW() 
             ORDER BY created_at DESC LIMIT 1",
            [$hwid]
        );
    }
    
    public function deactivateKey(string $key): bool {
        $result = $this->db->update('keys', ['is_active' => false], 'key_value = ?', [$key]);
        if ($result) {
            Logger::log('info', 'key_deactivated', "Key deactivated: " . substr($key, 0, 20));
        }
        return $result > 0;
    }
    
    public function extendKey(string $key, int $seconds): bool {
        $keyData = $this->db->fetch("SELECT expires_at FROM `keys` WHERE key_value = ?", [$key]);
        if (!$keyData) {

        $session = $this->getSession($sessionToken);
        
        if (!$session) {
            return ['success' => false, 'error' => 'Session not found or expired'];
        }

             WHERE session_id = ? AND checkpoint_number = ? AND verification_token = ?",

                'session' => substr($sessionToken, 0, 16),

            return ['success' => false, 'error' => 'Invalid verification'];
        }
        
        if ($checkpoint['completed']) {
            return [
                'success' => true,
                'already_completed' => true,
                'checkpoints_completed' => $session['checkpoints_completed'],
                'checkpoints_required' => $session['checkpoints_required']
            ];

                $session['ip_address'],

                $sessionToken,
                $checkpointNumber
            );

                    'reason' => $bypassCheck['reason'],
                    'session' => substr($sessionToken, 0, 16)
                ]);
                
                // Блокируем сессию
                $this->db->update('key_sessions', ['status' => 'blocked'], 'id = ?', [$session['id']]);
                
                return ['success' => false, 'error' => 'Verification failed'];
            }
        }

        $this->db->update('checkpoints', [
            'completed' => 1,
            'completed_at' => date('Y-m-d H:i:s')
        ], 'id = ?', [$checkpoint['id']]);
        
        // Обновляем счетчик в сессии
        $newCompletedCount = $session['checkpoints_completed'] + 1;
        $this->db->update('key_sessions', [
            'checkpoints_completed' => $newCompletedCount

        Logger::log('info', 'checkpoint_completed', "Checkpoint completed", [
            'session' => substr($sessionToken, 0, 16),
            'checkpoint' => $checkpointNumber
        ]);

            'success' => true,
            'checkpoints_completed' => $newCompletedCount,

    public function validateCheckpoint(string $originalIp, string $currentIp, string $sessionToken, int $checkpoint): array {

        // Мягкая проверка - логируем, но не блокируем сразу
        if ($originalIp !== $currentIp) {

        $referer = $_SERVER['HTTP_REFERER'] ?? '';
        if (!empty($referer)) {
            // Должен содержать linkvertise или наш домен
            $validReferers = ['linkvertise.', 'link-to.net', 'semyaware.ru'];
            $validReferer = false;
            
            foreach ($validReferers as $valid) {
                if (stripos($referer, $valid) !== false) {
                    $validReferer = true;
                    break;
                }
            }
            
            if (!$validReferer) {
                Logger::log('warning', 'invalid_referer', "Invalid referer detected", [
                    'referer' => $referer,
                    'session' => substr($sessionToken, 0, 16)
                ]);
                // Подозрительно, но не блокируем
            }
        }
        
        // Проверка 3: Время между чекпоинтами
        if ($checkpoint > 1) {
            $prevCompleted = $this->db->fetch(

                 JOIN key_sessions s ON c.session_id = s.id
                 WHERE s.session_token = ? AND c.checkpoint_number = ?",

                if ($timeDiff < 3) {

                    ]);

                }

        $attempts = $this->db->fetch(

            ]);

        Logger::log('debug', 'checkpoint_verification_attempt', "Checkpoint verification attempt", [
            'session' => substr($sessionToken, 0, 16),
            'checkpoint' => $checkpoint
        ]);
        
        return ['valid' => true];

 * Rate Limiting Protection

class RateLimiter {
    private $db;
    
    public function __construct() {
        $this->db = Database::getInstance();
    }
    
    public function check(string $identifier, string $endpoint, int $maxRequests = null, int $windowSeconds = null): array {
        $maxRequests = $maxRequests ?? RATE_LIMIT_REQUESTS;
        $windowSeconds = $windowSeconds ?? RATE_LIMIT_WINDOW;
        
        $record = $this->db->fetch(
            "SELECT * FROM rate_limits WHERE identifier = ? AND endpoint = ?",
            [$identifier, $endpoint]
        );
        
        $now = time();
        
        if (!$record) {
            // Первый запрос
            $this->db->insert('rate_limits', [
                'identifier' => $identifier,
                'endpoint' => $endpoint,
                'request_count' => 1,
                'window_start' => date('Y-m-d H:i:s', $now)
            ]);
            
            return [
                'allowed' => true,
                'remaining' => $maxRequests - 1,
                'reset_at' => $now + $windowSeconds
            ];
        }
        
        $windowStart = strtotime($record['window_start']);
        
        // Окно истекло - сбрасываем
        if ($now - $windowStart > $windowSeconds) {
            $this->db->update('rate_limits', [
                'request_count' => 1,
                'window_start' => date('Y-m-d H:i:s', $now)
            ], 'id = ?', [$record['id']]);
            
            return [
                'allowed' => true,
                'remaining' => $maxRequests - 1,
                'reset_at' => $now + $windowSeconds
            ];
        }
        
        // Проверяем лимит
        if ($record['request_count'] >= $maxRequests) {
            $resetAt = $windowStart + $windowSeconds;
            
            Logger::log('warning', 'rate_limited', "Rate limit exceeded", [
                'identifier' => $identifier,
                'endpoint' => $endpoint,
                'count' => $record['request_count']
            ]);
            
            return [
                'allowed' => false,
                'remaining' => 0,
                'reset_at' => $resetAt,
                'retry_after' => $resetAt - $now
            ];
        }
        
        // Увеличиваем счетчик
        $this->db->update('rate_limits', [
            'request_count' => $record['request_count'] + 1
        ], 'id = ?', [$record['id']]);
        
        return [
            'allowed' => true,
            'remaining' => $maxRequests - $record['request_count'] - 1,
            'reset_at' => $windowStart + $windowSeconds
        ];
    }
    
    public function cleanup(): int {
        return $this->db->query(
            "DELETE FROM rate_limits WHERE window_start < DATE_SUB(NOW(), INTERVAL 1 HOUR)"
        )->rowCount();
    }

            errorMessage: document.getElementById('error-message')
        };
        
        this.init();
    }
    
    init() {
        // Check URL parameters for callback
        this.handleCallback();
        
        // Event listeners
        this.elements.startBtn.addEventListener('click', () => this.startSession());
        this.elements.copyBtn.addEventListener('click', () => this.copyKey());
        this.elements.retryBtn.addEventListener('click', () => this.reset());
        
        // Checkpoint buttons
        document.querySelectorAll('.btn-checkpoint').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const cp = parseInt(e.target.dataset.checkpoint);
                this.openCheckpoint(cp);
            });
        });
        
        // Enter key on HWID input
        this.elements.hwidInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') this.startSession();
        });
        
        // Load saved session
        this.loadSession();
    }
    
    handleCallback() {
        const params = new URLSearchParams(window.location.search);
        
        if (params.has('verified')) {
            const session = params.get('session');
            const completed = parseInt(params.get('completed'));
            const total = parseInt(params.get('total'));
            
            if (session) {
                this.sessionToken = session;
                localStorage.setItem('keySessionToken', session);
                this.checkpointsCompleted = completed;
                this.checkpointsRequired = total;
                
                // Clear URL
                window.history.replaceState({}, '', window.location.pathname);
                
                // Show checkpoint section and update UI
                this.showCheckpointSection();
                this.updateProgress();
                
                if (completed >= total) {
                    this.getKey();
                }
            }
        }
        
        if (params.has('error')) {
            this.showError('Verification Failed', params.get('error'));
            window.history.replaceState({}, '', window.location.pathname);
        }
    }
    
    loadSession() {
        const savedSession = localStorage.getItem('keySessionToken');
        const savedHwid = localStorage.getItem('keyHwid');
        
        if (savedSession && savedHwid) {
            this.sessionToken = savedSession;
            this.hwid = savedHwid;
            this.elements.hwidInput.value = savedHwid;
            
            // Check session status
            this.checkSessionStatus();
        }
    }
    
    saveSession() {
        if (this.sessionToken) {
            localStorage.setItem('keySessionToken', this.sessionToken);
        }
        if (this.hwid) {
            localStorage.setItem('keyHwid', this.hwid);
        }
    }
    
    clearSession() {
        localStorage.removeItem('keySessionToken');
        localStorage.removeItem('keyHwid');
        this.sessionToken = null;
        this.hwid = null;
    }
    
    async checkSessionStatus() {
        try {
            const response = await fetch(`${CONFIG.apiBase}/info.php?session=${this.sessionToken}&hwid=${this.hwid}`);
            const data = await response.json();
            
            if (data.key && data.key.exists) {
                // Already has key
                this.showKey(data.key.key, data.key.expires_in);
                return;
            }
            
            if (data.session) {
                this.checkpointsCompleted = data.session.checkpoints_completed;
                this.checkpointsRequired = data.session.checkpoints_required;
                
                if (data.session.status === 'completed') {
                    this.getKey();
                } else if (data.session.status === 'pending') {
                    this.showCheckpointSection();
                    this.updateProgress();
                }
            }
        } catch (error) {
            console.error('Failed to check session:', error);
        }
    }
    
    async startSession() {
        const hwid = this.elements.hwidInput.value.trim();
        
        if (!hwid || hwid.length < 16) {
            this.showError('Invalid HWID', 'Please enter a valid HWID (at least 16 characters)');
            return;
        }
        
        this.hwid = hwid;
        this.showLoading('Starting session...');
        
        try {
            const response = await fetch(`${CONFIG.apiBase}/start.php`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ hwid })
            });
            
            const data = await response.json();
            
            if (!data.success) {
                throw new Error(data.error || 'Failed to start session');
            }
            
            if (data.has_key) {
                // Already has valid key
                this.showKey(data.key, data.expires_in);
                return;
            }
            
            this.sessionToken = data.session_token;
            this.checkpointsCompleted = data.checkpoints_completed;
            this.checkpointsRequired = data.checkpoints_required;
            
            this.saveSession();
            this.showCheckpointSection();
            this.updateProgress();
            
        } catch (error) {
            this.showError('Error', error.message);
        } finally {
            this.hideLoading();
        }
    }
    
    async openCheckpoint(checkpointNumber) {
        if (checkpointNumber > this.checkpointsCompleted + 1) {
            return; // Can't skip checkpoints
        }
        
        this.showLoading('Getting link...');
        
        try {
            const response = await fetch(`${CONFIG.apiBase}/checkpoint.php`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    session_token: this.sessionToken,
                    checkpoint: checkpointNumber
                })
            });
            
            const data = await response.json();
            
            if (!data.success) {
                throw new Error(data.error || 'Failed to get checkpoint link');
            }
            
            if (data.completed) {
                // Already completed
                this.markCheckpointComplete(checkpointNumber);
                return;
            }
            
            // Open Linkvertise link
            window.location.href = data.url;
            
        } catch (error) {
            this.showError('Error', error.message);
        } finally {
            this.hideLoading();
        }
    }
    
    async getKey() {
        this.showLoading('Generating key...');
        
        try {
            const response = await fetch(`${CONFIG.apiBase}/getkey.php`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    session_token: this.sessionToken
                })
            });
            
            const data = await response.json();
            
            if (!data.success) {
                throw new Error(data.error || 'Failed to generate key');
            }
            
            this.showKey(data.key, data.expires_in);
            this.clearSession();
            
        } catch (error) {
            this.showError('Error', error.message);
        } finally {
            this.hideLoading();
        }
    }
    
    showCheckpointSection() {
        this.hideAllSections();
        this.elements.checkpointSection.classList.remove('hidden');
        
        // Enable/disable checkpoint buttons
        for (let i = 1; i <= this.checkpointsRequired; i++) {
            const checkpoint = document.getElementById(`checkpoint-${i}`);
            const btn = checkpoint.querySelector('.btn-checkpoint');
            
            if (i <= this.checkpointsCompleted) {
                this.markCheckpointComplete(i);
            } else if (i === this.checkpointsCompleted + 1) {
                checkpoint.classList.add('active');
                checkpoint.querySelector('.checkpoint-status').textContent = 'Ready';
                btn.disabled = false;
            } else {
                checkpoint.classList.remove('active', 'completed');
                checkpoint.querySelector('.checkpoint-status').textContent = 'Pending';
                btn.disabled = true;
            }
        }
    }
    
    markCheckpointComplete(checkpointNumber) {
        const checkpoint = document.getElementById(`checkpoint-${checkpointNumber}`);
        checkpoint.classList.remove('active');
        checkpoint.classList.add('completed');
        checkpoint.querySelector('.checkpoint-status').textContent = 'Completed';
        checkpoint.querySelector('.btn-checkpoint').disabled = true;
    }
    
    updateProgress() {
        const percent = (this.checkpointsCompleted / this.checkpointsRequired) * 100;
        this.elements.progressFill.style.width = `${percent}%`;
        this.elements.progressText.textContent = `Step ${this.checkpointsCompleted} of ${this.checkpointsRequired}`;
        
        // Check if all completed
        if (this.checkpointsCompleted >= this.checkpointsRequired) {
            this.getKey();
        }
    }
    
    showKey(key, expiresIn) {
        this.hideAllSections();
        this.elements.keySection.classList.remove('hidden');
        this.elements.keyValue.value = key;
        this.elements.keyExpires.textContent = this.formatDuration(expiresIn);
    }
    
    showError(title, message) {
        this.hideAllSections();
        this.elements.errorSection.classList.remove('hidden');
        this.elements.errorTitle.textContent = title;
        this.elements.errorMessage.textContent = message;
    }
    
    hideAllSections() {
        this.elements.hwidSection.classList.add('hidden');
        this.elements.checkpointSection.classList.add('hidden');
        this.elements.keySection.classList.add('hidden');
        this.elements.errorSection.classList.add('hidden');
        this.hideLoading();
    }
    
    showLoading(text = 'Loading...') {
        this.elements.loadingText.textContent = text;
        this.elements.loading.classList.remove('hidden');
    }
    
    hideLoading() {
        this.elements.loading.classList.add('hidden');
    }
    
    reset() {
        this.clearSession();
        this.checkpointsCompleted = 0;
        this.hideAllSections();
        this.elements.hwidSection.classList.remove('hidden');
        this.elements.hwidInput.value = '';
        
        // Reset checkpoints UI
        for (let i = 1; i <= this.checkpointsRequired; i++) {
            const checkpoint = document.getElementById(`checkpoint-${i}`);
            if (checkpoint) {
                checkpoint.classList.remove('active', 'completed');
                checkpoint.querySelector('.checkpoint-status').textContent = 'Pending';
                checkpoint.querySelector('.btn-checkpoint').disabled = true;
                checkpoint.querySelector('.checkpoint-number').style.display = '';
                checkpoint.querySelector('.check-icon').classList.add('hidden');
            }
        }
        
        this.elements.progressFill.style.width = '0%';
    }
    
    copyKey() {
        const key = this.elements.keyValue.value;
        navigator.clipboard.writeText(key).then(() => {
            const btn = this.elements.copyBtn;
            btn.innerHTML = '<svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><path d="M5 13l4 4L19 7"/></svg>';
            btn.style.background = 'var(--success)';
            
            setTimeout(() => {
                btn.innerHTML = '<svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><rect x="9" y="9" width="13" height="13" rx="2"/><path d="M5 15H4a2 2 0 01-2-2V4a2 2 0 012-2h9a2 2 0 012 2v1"/></svg>';
                btn.style.background = '';
            }, 2000);
        });
    }
    
    formatDuration(seconds) {
        const hours = Math.floor(seconds / 3600);
        const minutes = Math.floor((seconds % 3600) / 60);
        
        if (hours > 24) {
            const days = Math.floor(hours / 24);
            return `${days} day${days > 1 ? 's' : ''}`;
        }
        
        if (hours > 0) {
            return `${hours}h ${minutes}m`;
        }
        
        return `${minutes} minutes`;
    }
}

local function generateHWID()
    local player = Players.LocalPlayer
    local hwid_parts = {
        tostring(player.UserId),
        tostring(game.PlaceId),
        tostring(game.GameId),
        game:GetService("RbxAnalyticsService"):GetClientId()
    }
    
    -- Simple hash
    local combined = table.concat(hwid_parts, "-")
    local hash = 0
    for i = 1, #combined do
        hash = bit32.bxor(hash * 31, string.byte(combined, i))
    end
    
    return string.format("%016X", hash) .. "-" .. tostring(player.UserId)
end

-- HTTP Request wrapper with retry
local function httpRequest(url, method, body)
    method = method or "GET"
    
    for attempt = 1, CONFIG.RETRY_ATTEMPTS do
        local success, response = pcall(function()
            if method == "GET" then
                return game:HttpGet(url)
            else
                return (syn and syn.request or http_request or request)({
                    Url = url,
                    Method = method,
                    Headers = {
                        ["Content-Type"] = "application/json"
                    },
                    Body = body and HttpService:JSONEncode(body) or nil
                })
            end
        end)
        
        if success then
            if type(response) == "string" then
                return HttpService:JSONDecode(response)
            elseif response.Body then
                return HttpService:JSONDecode(response.Body)
            end
        end
        
        if attempt < CONFIG.RETRY_ATTEMPTS then
            wait(1)
        end
    end
    
    return nil, "Request failed after " .. CONFIG.RETRY_ATTEMPTS .. " attempts"

-- Validate key with API
function KeySystem:ValidateKey(key)
    local url = CONFIG.API_URL .. "/validatekey.php?key=" .. key .. "&hwid=" .. self.HWID
    
    local response, err = httpRequest(url, "GET")
    
    if not response then
        return false, err or "Validation request failed"
    end
    
    if response.valid then
        self.Key = key
        self.Validated = true
        self:SaveKeyToFile(key)
        
        return true, {
            expires_in = response.expires_in,
            expires_at = response.expires_at,
            is_premium = response.is_premium
        }

        return false, response.error or "Invalid key"
    end
end
